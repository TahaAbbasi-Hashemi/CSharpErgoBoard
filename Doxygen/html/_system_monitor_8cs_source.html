359  {
360  m_computer.CPUEnabled = true;
361 
362  // Gets the total amount of CPU cores.
363  for (Int32 i = 0; i < m_computer.Hardware.Length; i++)
364  {
365  // Pick CPU hardware
366  if (m_computer.Hardware[i].HardwareType != HardwareType.CPU)
367  {
368  continue;
369  }
370  for (Int32 j = 0; j < m_computer.Hardware[i].Sensors.Length; j++)
371  {
372  // Pick random CPU sensor. All cores should have a temp.
373  if (m_computer.Hardware[i].Sensors[j].SensorType != SensorType.Temperature)
